23     @Transactional

29             Delivery newDelivery = new Delivery(
30                 id: params.deliveryId,
31                 requestedAt: params.requestedAt,
32                 status: 'CREATED'
33             )
34             saveDeliveryPort.saveDelivery(newDelivery)
35         }
36     }
37 
38     void logDeliveryIsAlreadyPresent(Delivery delivery) {
39         log.debug("delivery ${delivery.id} is already present")
40     }
