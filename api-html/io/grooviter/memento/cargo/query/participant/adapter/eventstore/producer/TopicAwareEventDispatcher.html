02 
03 import com.fasterxml.jackson.databind.ObjectMapper
04 import io.grooviter.memento.cargo.query.participant.adapter.eventstore.events.Created
05 import io.grooviter.memento.cargo.shared.TopicAwareEvent
06 import io.micronaut.context.event.ApplicationEventListener
07 import io.micronaut.context.event.ApplicationEventPublisher
08 
09 import javax.inject.Inject
10 import javax.inject.Singleton
11 
12 @Singleton
13 class TopicAwareEventDispatcher implements ApplicationEventListener<TopicAwareEvent> {
14 
15     @Inject
16     ApplicationEventPublisher applicationEventPublisher
17 
18     @Inject
19     ObjectMapper objectMapper
20 
21     @Override
22     void onApplicationEvent(TopicAwareEvent event) {
23         switch (event.topic) {
24             case 'PARTICIPANT_REGISTERED': dispatchParticipantRegistered(event); break
25         }
26     }
27 
28     private dispatchParticipantRegistered(TopicAwareEvent event) {
29         Created created = objectMapper
30             .readerFor(Created)
31             .readValue(event.json)
32 
33         applicationEventPublisher.publishEvent(created)
34     }
35 }
