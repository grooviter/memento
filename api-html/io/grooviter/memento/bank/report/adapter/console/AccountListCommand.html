AccountListCommand.groovy
01 package io.grooviter.memento.bank.report.adapter.console
02 
03 import groovy.transform.CompileStatic
04 import io.grooviter.memento.bank.infra.console.AllCommands
05 import io.grooviter.memento.bank.infra.console.Utils
06 import io.grooviter.memento.bank.report.application.port.in.UseCases
07 import io.grooviter.memento.bank.report.domain.Account
08 import picocli.CommandLine
09 
10 import javax.inject.Inject
11 
12 @CompileStatic
13 @CommandLine.Command(version = "1.0", name = "account-list", description = "lists all available accounts")
14 class AccountListCommand implements Runnable {
15     @CommandLine.ParentCommand
16     AllCommands parent
17 
18     @Inject
19     UseCases.ListAccountsQuery listAccountsQuery
20 
21     @Override
22     void run() {
23         List<Account> accountList = listAccountsQuery.list()
24 
25         if (!accountList) {
26             parent.printlnAnsi("@|yellow NO ACCOUNTS FOUND |@")
27             return
28         }
29 
30         parent.printlnAnsi(createTable(accountList))
31     }
32 
33     private static String createTable(List<Account> accountList) {
34         return Utils.tableBuilder(Account)
35             .columnNames('ID', 'HOLDER NAME', 'CREATED_AT')
36             .addColumnConsumer((account) -> account.id)
37             .addColumnConsumer((account) -> account.name)
38             .addColumnConsumer((account) -> account.createdAt)
39             .rows(accountList)
40             .build()
41     }
42 }
