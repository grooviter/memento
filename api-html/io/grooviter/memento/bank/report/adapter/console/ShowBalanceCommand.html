ShowBalanceCommand.groovy
01 package io.grooviter.memento.bank.report.adapter.console
02 
03 import groovy.transform.CompileStatic
04 import io.grooviter.memento.bank.infra.console.AllCommands
05 import io.grooviter.memento.bank.infra.console.Utils
06 import io.grooviter.memento.bank.report.application.port.in.UseCases
07 import io.grooviter.memento.bank.report.domain.Balance
08 import picocli.CommandLine
09 
10 import javax.inject.Inject
11 
12 @CompileStatic
13 @CommandLine.Command(version = "1.0", name = "account-balance", description = "shows a given account balance")
14 class ShowBalanceCommand implements Runnable {
15     @CommandLine.ParentCommand
16     AllCommands parent
17 
18     @CommandLine.Option(names = ['-i', '--account'], description = "account ID", required = true)
19     UUID accountId
20 
21     @Inject
22     UseCases.ShowAccountBalanceQuery showAccountBalanceQuery
23 
24     @Override
25     void run() {
26         UseCases.ShowAccountBalanceQuery.Params params = UseCases.ShowAccountBalanceQuery.Params.builder()
27             .accountId(accountId)
28             .build()
29 
30         showAccountBalanceQuery
31             .showBalance(params)
32             .map(ShowBalanceCommand::renderBalance)
33             .ifPresentOrElse(
34                 message -> parent.printlnAnsi(message),
35                 () -> parent.printlnAnsi("@|yellow NO BALANCE FOUND |@"))
36     }
37 
38     private static String renderBalance(Balance balance) {
39         return Utils.tableBuilder(Balance)
40             .columnNames('DATE', 'BALANCE')
41             .addColumnConsumer(blc -> blc.createdAt)
42             .addColumnConsumer(blc -> blc.balance)
43             .setTableWidth(60)
44             .rows([balance])
45             .build()
46     }
47 }
