ReportPersistenceAdapter.groovy
01 package io.grooviter.memento.bank.report.adapter.persistence
02 
03 import io.grooviter.memento.bank.infra.Configuration
04 import io.grooviter.memento.bank.infra.persistence.Utils
05 import io.grooviter.memento.bank.report.application.port.out.ReportPorts
06 import io.grooviter.memento.bank.report.domain.Account
07 import io.grooviter.memento.bank.report.domain.Balance
08 
09 import javax.inject.Inject
10 import javax.inject.Singleton
11 import java.time.OffsetDateTime
12 import java.util.function.Predicate
13 import java.util.stream.Collectors
14 
15 @Singleton
16 class ReportPersistenceAdapter implements
17     ReportPorts.ListAccountPort,
18     ReportPorts.LoadBalancePort,
19     ReportPorts.ListBalancePort {
20 
21     @Inject
22     Configuration configuration
23 
24     @Override
25     List<Account> list() {
26         return Utils
27             .createStreamForFile(configuration.projections.accounts, Account)
28             .collect(Collectors.toList())
29     }
30 
31     @Override
32     Optional<Balance> loadBalance(UUID accountId) {
33         return Utils
34             .createStreamForFile(configuration.projections.balances, Balance)
35             .filter(idEquals(accountId))
36             .sorted(byDateDesc())
37             .findFirst()
38     }
39 
40     @Override
41     List<Balance> listBalanceHistory(UUID accountId, OffsetDateTime from) {
42         return Utils
43             .createStreamForFile(configuration.projections.balances, Balance)
44             .filter(idEquals(accountId))
45             .sorted(byDateDesc())
46             .filter(afterDate(from))
47             .collect(Collectors.toList())
48     }
49 
50     private static Predicate<Balance> afterDate(OffsetDateTime from) {
51         return (balance) -> Optional
52             .ofNullable(from)
53             .map(date -> balance.createdAt.isAfter(date))
54             .orElse(true)
55     }
56 
57     private static Predicate<Balance> idEquals(UUID id) {
58         return (balance) -> balance.accountId == id.toString()
59     }
60 
61     private static Comparator<Balance> byDateDesc() {
62         return (left, right) -> left.createdAt.isAfter(right.createdAt) ? -1 : 1
63     }
64 }
