MicronautFactoryInitializer.groovy
01 package io.grooviter.memento.bank.infra.console
02 
03 import io.micronaut.configuration.picocli.MicronautFactory
04 import io.micronaut.context.ApplicationContext
05 import io.micronaut.context.ApplicationContextBuilder
06 import io.micronaut.context.env.CommandLinePropertySource
07 import io.micronaut.context.env.Environment
08 import io.micronaut.core.cli.CommandLine
09 
10 class MicronautFactoryInitializer {
11     static <R extends Runnable> MicronautFactory initialize(Class<R> commandClazz, String... args) {
12         ApplicationContextBuilder builder = buildApplicationContext(commandClazz, args)
13         try (ApplicationContext ctx = builder.start()) {
14             return new MicronautFactory(ctx)
15         }
16     }
17 
18     private static ApplicationContextBuilder buildApplicationContext(Class<?> cls, String[] args) {
19         CommandLine commandLine = CommandLine.parse(args)
20         CommandLinePropertySource commandLinePropertySource = new CommandLinePropertySource(commandLine)
21         return ApplicationContext
22                 .builder(cls, Environment.CLI)
23                 .propertySources(commandLinePropertySource)
24     }
25 }
