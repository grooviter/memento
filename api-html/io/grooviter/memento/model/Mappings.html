03 class Mappings {

05     final Map<String, Class> aliasToType

09     Mappings(List<Mapping> list) {

11             .asUnmodifiable()
12         this.aliasToType = list

14             .asUnmodifiable()
15         this.typeToAlias = list

17             .asUnmodifiable()
18     }
19 
20     Class resolveClass(String alias){
21         Optional<Class> resolvedClass = Optional.ofNullable(this.aliasToType.get(alias))
22                 | () -> Optional.of(Class.forName(alias)) // only executed if doesn't have an alias
23 
24         return resolvedClass.orElse(null)
25     }
26 
27     String resolveAlias(Class type) {
28         return this.typeToAlias.get(type) ?: type.name
29     }
30 
31     static Builder builder() {

42
