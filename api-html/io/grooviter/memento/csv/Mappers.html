Mappers.groovy
01 package io.grooviter.memento.csv
02 
03 import io.grooviter.memento.EventSerdePort
04 import io.grooviter.memento.model.Aggregate
05 import io.grooviter.memento.model.Event
06 
07 import java.time.OffsetDateTime
08 
09 class Mappers {
10 
11     static final String SEPARATOR_SPLIT = '\\|'
12     static final String SEPARATOR_JOIN = '|'
13     static final String NEWLINE = '\n'
14 
15     static Aggregate toAggregate(String row, EventSerdePort serdePort) {
16         def (id, aggregateId, aggregateType, version, json, createdAt) = row.split(SEPARATOR_SPLIT)
17 
18         return serdePort.aggregateFromJSON("$aggregateType", "$json")
19     }
20 
21     static Event toEvent(String row, EventSerdePort serdePort) {
22         def (id, aggregateId, version, type, json, createdAt) = row.split(SEPARATOR_SPLIT)
23 
24         return serdePort.eventFromJSON("$type", "$json")
25     }
26 
27     static String toSnapshotRow(Aggregate aggregate, EventSerdePort serdePort) {
28         Map<String, ?> rowMap = [
29             id: UUID.randomUUID(),
30             aggregateId: aggregate.id,
31             aggregateType: serdePort.resolveAlias(aggregate.getClass()),
32             version: aggregate.version,
33             json: serdePort.aggregateToJSON(aggregate),
34             createdAt: serdePort.objectToJSON(OffsetDateTime.now())
35         ]
36 
37         rowMap.values()*.toString().join(SEPARATOR_JOIN) + NEWLINE
38     }
39 
40     static String toEventRow(Event event, EventSerdePort serdePort) {
41         Map<String, ?> rowMap = [
42             id: event.id,
43             aggregateId: event.aggregateId,
44             version: event.version,
45             type: serdePort.resolveAlias(event.getClass()),
46             json: serdePort.eventToJSON(event),
47             createdAt: serdePort.objectToJSON(event.createdAt)
48         ]
49 
50         rowMap.values()*.toString().join(SEPARATOR_JOIN) + NEWLINE
51     }
52 }
