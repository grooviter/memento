CsvEventStorage.groovy
01 package io.grooviter.memento.csv
02 
03 import io.grooviter.memento.EventStoragePort
04 import io.grooviter.memento.EventSerdePort
05 import io.grooviter.memento.model.Aggregate
06 import io.grooviter.memento.model.Event
07 import groovy.transform.TupleConstructor
08 
09 import java.nio.file.Files
10 import java.util.function.Predicate
11 import java.util.stream.Stream
12 
13 @TupleConstructor
14 class CsvEventStorage implements EventStoragePort {
15 
16     static final String SEPARATOR = '\\|'
17 
18     static final Integer ROW_COLUMN_AGGREGATE_ID = 1
19     static final Integer ROW_COLUMN_EVENT_TYPE = 3
20 
21     File eventStorageFile
22     File snapshotStorageFile
23 
24     static CsvEventStorage create(File events, File snapshots) {
25         if (!events.exists()) {
26             events.createNewFile()
27         }
28         if (!snapshots.exists()) {
29             snapshots.createNewFile()
30         }
31 
32         return new CsvEventStorage(events, snapshots)
33     }
34 
35     @Override
36     void append(Event event, EventSerdePort serdePort) {
37         eventStorageFile.append(Mappers.toEventRow(event, serdePort))
38     }
39 
40     @Override
41     void snapshot(Aggregate aggregate, EventSerdePort serdePort) {
42         snapshotStorageFile.append(Mappers.toSnapshotRow(aggregate, serdePort))
43     }
44 
45     @Override
46     Optional<Aggregate> findByAggregateIdOrderByVersionDesc(UUID id, EventSerdePort serdePort) {
47         return Files.lines(snapshotStorageFile.toPath())
48             .filter(byAggregateId(id))
49             .map(line -> Mappers.toAggregate(line, serdePort))
50             .sorted { left, right -> right.version - left.version }
51             .limit(1)
52             .findFirst()
53     }
54 
55     @Override
56     Stream<Event> findAllByAggregateIdAndVersionGreaterThanOrderByVersion(UUID aggregateId, Integer version, EventSerdePort serdePort) {
57         return Files.lines(eventStorageFile.toPath())
58             .filter(byAggregateId(aggregateId))
59             .map(row -> Mappers.toEvent(row, serdePort))
60             .filter(byVersionGreaterEqual(version))
61     }
62 
63     private static Predicate<String> byAggregateId(UUID id) {
64         return (String row) -> row.split(SEPARATOR)[ROW_COLUMN_AGGREGATE_ID] == id.toString()
65     }
66 
67     private static Predicate<Event> byVersionGreaterEqual(Integer version) {
68         return (Event event) -> event.version >= version
69     }
70 
71     @Override
72     Stream<Event> findAllByAggregateId(UUID aggregateId, EventSerdePort serdePort) {
73         return Files.lines(eventStorageFile.toPath())
74             .filter(row -> row.split(SEPARATOR)[ROW_COLUMN_AGGREGATE_ID] == aggregateId.toString())
75             .map(row -> Mappers.toEvent(row, serdePort))
76     }
77 
78     @Override
79     Stream<Event> findAllByAliases(String[] aliases, EventSerdePort serdePort) {
80         return Files.lines(eventStorageFile.toPath())
81             .filter(row -> row.split(SEPARATOR)[ROW_COLUMN_EVENT_TYPE] in aliases)
82             .map(row -> Mappers.toEvent(row, serdePort))
83     }
84 }
