JacksonEventSerde.groovy
01 package io.grooviter.memento.impl
02 
03 import com.fasterxml.jackson.databind.ObjectMapper
04 import com.fasterxml.jackson.databind.SerializationFeature
05 import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule
06 import groovy.util.logging.Slf4j
07 import io.grooviter.memento.EventSerdePort
08 import io.grooviter.memento.model.Aggregate
09 import io.grooviter.memento.model.Event
10 import io.grooviter.memento.model.Mappings
11 
12 @Slf4j
13 class JacksonEventSerde implements EventSerdePort {
14 
15     private final Mappings mappings
16     private final ObjectMapper mapper
17 
18     JacksonEventSerde(List<Mappings> mappings, ObjectMapper objectMapper) {
19         this.mappings = Optional
20             .ofNullable(mappings)
21             .map(mappings1 -> mappings1.collectMany { it.mappingList })
22             .map(Mappings::new)
23             .orElse(Mappings.builder().build())
24 
25         this.mapper = Optional
26             .ofNullable(objectMapper)
27             .orElse(defaultObjectMapper())
28     }
29 
30     JacksonEventSerde() {
31         this.mappings = Mappings.builder().build()
32         this.mapper = defaultObjectMapper()
33     }
34 
35     private static ObjectMapper defaultObjectMapper() {
36         return new ObjectMapper()
37                 .registerModule(new JavaTimeModule())
38                 .disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS)
39     }
40 
41     @Override
42     String aggregateToJSON(Aggregate aggregate) {
43         return this.mapper.writeValueAsString(aggregate)
44     }
45 
46     @Override
47     String eventToJSON(Event event) {
48         return this.mapper.writeValueAsString(event)
49     }
50 
51     @Override
52     String objectToJSON(Object o) {
53         return this.mapper.writeValueAsString(o)
54     }
55 
56     @Override
57     Aggregate aggregateFromJSON(String alias, String json) {
58         Class clazz = this.mappings.resolveClass(alias)
59         JacksonEventSerde.log.debug("clazz for $alias is ${clazz.simpleName}")
60         return this.mapper.readValue(json, clazz)
61     }
62 
63     @Override
64     Event eventFromJSON(String alias, String json) {
65         return this.mapper.readValue(json, this.mappings.resolveClass(alias))
66     }
67 
68     @Override
69     String resolveAlias(Class clazz) {
70         String alias = this.mappings.resolveAlias(clazz)
71         JacksonEventSerde.log.debug("alias for ${clazz.simpleName} is $alias")
72         return alias
73     }
74 }
