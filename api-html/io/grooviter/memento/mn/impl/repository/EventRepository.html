11 /**

16 @Repository

18 abstract class EventRepository implements PageableRepository<EventEntity, UUID> {
19 
20     /**
21      * Alias for saving a new event to the databae

28     }
29 
30     /**
31      * Retrieves all events from a given aggregate from the the beginning
32      *
33      * @param aggregateId aggregate we want the events from
34      * @return an {@link Stream} of {@link EventEntity} instances
35      * @since 0.1.0
36      */
37     abstract Stream<EventEntity> findAllByAggregateId(UUID aggregateId)
38 
39     /**
40      * Retrieves all events from a given aggregate from a specific version
41      *
42      * @param aggregateId aggregate we want the events from
43      * @param version initial point
44      * @return an {@link Stream} of {@link EventEntity} instances
45      * @since 0.1.0
46      */
47     abstract Stream<EventEntity> findAllByAggregateIdAndVersionGreaterThanOrderByVersion(
48         UUID aggregateId,
49         Integer version)
50 
51     /**
52      * Retrieves all events of a given type
53      *
54      * @param aliases list with names of the type of the events
55      * @return an stream of events of the types passed as arguments
56      * @since 0.1.0
57      */
58     abstract Stream<EventEntity> findAllByTypeIn(String[] aliases)
59 }
