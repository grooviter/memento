Mappers.groovy
01 package io.grooviter.memento.mn.impl
02 
03 import io.grooviter.memento.EventSerdePort
04 import io.grooviter.memento.mn.impl.entity.EventEntity
05 import io.grooviter.memento.mn.impl.entity.SnapshotEntity
06 import io.grooviter.memento.model.Aggregate
07 import io.grooviter.memento.model.Event
08 
09 /**
10  * Utility class to convert from domain to entity objects
11  *
12  * @since 0.1.0
13  */
14 class Mappers {
15 
16     static <T extends Event> T toEvent(EventSerdePort serdePort, EventEntity source) {
17         return serdePort.eventFromJSON(source.type, source.json)
18     }
19 
20     static <T extends Aggregate> T toAggregate(EventSerdePort serdePort, SnapshotEntity source) {
21         return serdePort.aggregateFromJSON(source.aggregateType, source.json) as T
22     }
23 
24     static EventEntity toEventEntity(EventSerdePort serdePort, Event event) {
25         return new EventEntity(
26             id: UUID.randomUUID(),
27             aggregateId: event.aggregateId,
28             version: event.version,
29             type: serdePort.resolveAlias(event.getClass()),
30             json: serdePort.eventToJSON(event)
31         )
32     }
33 
34     static SnapshotEntity toSnapshotEntity(EventSerdePort serdePort, Aggregate aggregate) {
35         return new SnapshotEntity(
36             id: UUID.randomUUID(),
37             aggregateId: aggregate.id,
38             aggregateType: serdePort.resolveAlias(aggregate.getClass()),
39             json: serdePort.aggregateToJSON(aggregate),
40             version: aggregate.version
41         )
42     }
43 }
