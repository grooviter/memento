09 //@Aggregate(events = [DocumentCreated, DocumentDeleted, ContentModified])
10 class Document extends Aggregate {
11     String content = "", title = "untitled", author = "unknown"
12 
13     static Document create(String title, String author) {
14         Document document = new Document(id: UUID.randomUUID())
15         return document.apply(Events.created(title, author, document.nextVersion))
16     }
17 
18     Document append(String content) {
19         return this.apply(Events.modified("${this.content} $content", nextVersion))
20     }
21 
22     Document delete() {
23         return this.apply(Events.deleted(nextVersion))
24     }
25 
26     Document apply(DocumentCreated documentCreated) {
27         super.apply(documentCreated)
28         this.title = documentCreated.title
29         this.author = documentCreated.author
30         return this
31     }
32 
33     Document apply(ContentModified contentModified) {
34         super.apply(contentModified)
35         this.content = contentModified.content
36         return this
37     }
38 
39     Document apply(DocumentDeleted documentDeleted) {
40         super.apply(documentDeleted)
41         return this
42     }
43 }
