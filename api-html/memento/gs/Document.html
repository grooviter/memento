01 package memento.gs
02 
03 import io.grooviter.memento.model.Aggregate
04 import io.grooviter.memento.model.Event
05 
06 /**
07  * Represents a text document
08  *
09  * @since 0.1.0
10  */
11 class Document extends Aggregate {
12     String title, author, content = ''
13 
14     static class Created extends Event {
15         String title, author
16     }
17 
18     static class Appended extends Event {
19         String content
20     }
21 
22     static Document create(String title, String author) {
23         Document document = new Document(id: UUID.randomUUID())
24         return document.apply(new Created(title: title, author: author, version: document.nextVersion))
25     }
26 
27     Document append(String content) {
28         return this.apply(new Appended(content: content, version: nextVersion))
29     }
30 
31     private Document apply(Created created) {
32         super.apply(created)
33         this.title = created.title
34         this.author = created.author
35         return this
36     }
37 
38     private Document apply(Appended appended) {
39         super.apply(appended)
40         this.content += appended.content
41         return this
42     }
43 }
